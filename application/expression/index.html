infixExpression = infixExpression + "#";
  Stack optr = new Stack();     // 保存操作符的栈
  optr.push('#'); // 首先让'#'入栈
  Stack opnd = new Stack();// 保存操作数的栈
  for (int i = 0; i < infixExpression.length(); i++) {
    if (isOperand(infixExpression.charAt(i))) {// 如果当前字符是一个数字
      opnd.push(Integer.parseInt(infixExpression.charAt(i) + ""));
    } else {// 当前字符是一个操作符
        switch (infixExpression.charAt(i)) {
            case '(':// 由于左括号的优先级最高，所以肯定入栈
                optr.push('(');
                break;
            case ')':// 如果碰到右括号，则操作符栈一直弹栈，直到遇到匹配的左括号为止
                while (optr.peek() != '(') {/* 每当弹出一个操作符时， 从操作数栈中
                弹出两个数进行相应数值运算，并将结果压入栈中*/
                    opnd.push(operate(optr.pop(), opnd));}
                optr.pop();// 弹出'('
                break;
            default:// 默认情况下:+ - * /
                while (optr.peek() != '#' &&
                compare(optr.peek(), infixExpression.charAt(i))) {
                    opnd.push(operate(optr.pop(), opnd));}
                optr.push(infixExpression.charAt(i));
            break;}}}
 return opnd.peek();
